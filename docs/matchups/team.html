Advantage Summary

adv_df <- data.frame(
  Category  = c(pts_label, opp_label, diff_label, "Win %"),
  Team_1    = c(s1$pts, s1$opp_pts, s1$pt_diff, s1$win_pct),
  Team_2    = c(s2$pts, s2$opp_pts, s2$pt_diff, s2$win_pct),
  Lower_Better = c(FALSE, TRUE, FALSE, FALSE)
) %>%
  mutate(
    Advantage = case_when(
      !Lower_Better & Team_1 > Team_2 ~ team_1,
      !Lower_Better & Team_2 > Team_1 ~ team_2,
      Lower_Better  & Team_1 < Team_2 ~ team_1,
      Lower_Better  & Team_2 < Team_1 ~ team_2,
      TRUE ~ "Even"
    )
  ) %>%
  select(Category, Advantage)

t1_adv <- sum(adv_df$Advantage == team_1)
t2_adv <- sum(adv_df$Advantage == team_2)
